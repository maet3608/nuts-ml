17PICK = 0.1 # Pick 10% of the data for a quick trial 

18NUM_EPOCHS = 20

24def load_samples(): 

25 from keras.datasets import cifar10 

26 (x_train, y_train), (x_test, y_test) = cifar10.load_data() 

27 return list(zip(x_train, map(int, y_train))), list(zip(x_test, map(int, y_test)))

30def load_names(): 

31 from keras.utils.data_utils import get_file 

32 dirname = 'cifar-10-batches-py' 

33 origin = 'http://www.cs.toronto.edu/~kriz/cifar-10-python.tar.gz' 

34 path = get_file(dirname, origin=origin, untar=True) 

35 with open(osp.join(path, 'batches.meta'), 'rb') as f: 

36 return pickle.load(f)['label_names']

39def create_network(): 

40 from keras.models import Sequential 

41 from keras.layers import Dense, Dropout, Activation, Flatten 

42 from keras.layers import Convolution2D, MaxPooling2D 

43 

44 model = Sequential() 

45 model.add(Convolution2D(32, (3, 3), padding='same', 

46 input_shape=INPUT_SHAPE)) 

47 model.add(Activation('relu')) 

48 model.add(Convolution2D(32, (3, 3))) 

49 model.add(Activation('relu')) 

50 model.add(MaxPooling2D(pool_size=(2, 2))) 

51 model.add(Dropout(0.5)) 

52 

53 model.add(Convolution2D(64, (3, 3), padding='same')) 

54 model.add(Activation('relu')) 

55 model.add(Convolution2D(64, (3, 3))) 

56 model.add(Activation('relu')) 

57 model.add(MaxPooling2D(pool_size=(2, 2))) 

58 model.add(Dropout(0.5)) 

59 

60 model.add(Flatten()) 

61 model.add(Dense(512)) 

62 model.add(Activation('relu')) 

63 model.add(Dropout(0.5)) 

64 model.add(Dense(NUM_CLASSES)) 

65 model.add(Activation('softmax')) 

66 

67 model.compile(loss='categorical_crossentropy', 

68 optimizer='adam', metrics=['accuracy']) 

69 return KerasNetwork(model, 'weights_cifar10.hd5')

72def train(): 

73 from keras.metrics import categorical_accuracy 

74 

75 rerange = TransformImage(0).by('rerange', 0, 255, 0, 1, 'float32') 

76 build_batch = (BuildBatch(BATCH_SIZE) 

77 .input(0, 'image', 'float32') 

78 .output(1, 'one_hot', 'uint8', NUM_CLASSES)) 

79 p = 0.1 

80 augment = (AugmentImage(0) 

81 .by('identical', 1.0) 

82 .by('elastic', p, [5, 5], [100, 100], [0, 100]) 

83 .by('brightness', p, [0.7, 1.3]) 

84 .by('color', p, [0.7, 1.3]) 

85 .by('shear', p, [0, 0.1]) 

86 .by('fliplr', p) 

87 .by('rotate', p, [-10, 10])) 

88 plot_eval = PlotLines((0, 1), layout=(2, 1)) 

89 

90 print('creating network...') 

91 network = create_network() 

92 

93 print('loading data...') 

94 train_samples, test_samples = load_samples() 

95 train_samples, val_samples = train_samples >> SplitRandom(0.8) 

96 

97 print('training...', len(train_samples), len(val_samples)) 

98 for epoch in range(NUM_EPOCHS): 

99 print('EPOCH:', epoch) 

100 

101 t_loss, t_acc = (train_samples >> PrintProgress(train_samples) >> 

102 Pick(PICK) >> augment >> rerange >> Shuffle(100) >> 

103 build_batch >> network.train() >> Unzip()) 

104 t_loss, t_acc = t_loss >> Mean(), t_acc >> Mean() 

105 print("train loss : {:.6f}".format(t_loss)) 

106 print("train acc : {:.1f}".format(100 * t_acc)) 

107 

108 v_loss, v_acc = (val_samples >> rerange >> 

109 build_batch >> network.validate() >> Unzip()) 

110 v_loss, v_acc = v_acc >> Mean(), v_acc >> Mean() 

111 print('val loss : {:.6f}'.format(v_loss)) 

112 print('val acc : {:.1f}'.format(100 * v_acc)) 

113 

114 network.save_best(v_acc, isloss=False) 

115 plot_eval((t_acc, v_acc)) 

116 

117 print('testing...', len(test_samples)) 

118 e_acc = (test_samples >> rerange >> build_batch >> 

119 network.evaluate([categorical_accuracy])) 

120 print('test acc : {:.1f}'.format(100 * e_acc))

123if __name__ == "__main__": 

124 train()
