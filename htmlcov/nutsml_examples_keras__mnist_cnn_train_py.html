1""" 

2.. module:: cnn_train 

3 :synopsis: Example nuts-ml pipeline for training a CNN on MNIST 

4 

5This is code is based on a Keras example (see here) 

6https://github.com/fchollet/keras/blob/master/examples/mnist_cnn.py 

7to train a Multi-layer perceptron on the MNIST data and modified to 

8use nuts for the data-preprocessing. 

9""" 

10 

11from __future__ import print_function 

12 

13from six.moves import zip, range 

14from nutsflow import PrintProgress, Collect, Unzip, Shuffle, Pick, Mean, NOP 

15from nutsml import (KerasNetwork, TransformImage, AugmentImage, BuildBatch, 

16 Boost, PrintColType, PlotLines) 

17 

18PICK = 0.1 # Pick 10% of the data for a quick trial 

19NUM_EPOCHS = 10 

20INPUT_SHAPE = (28, 28, 1) 

21BATCH_SIZE = 128 

22NUM_CLASSES = 10 

23 

24 

25def load_samples(): 

26 from keras.datasets import mnist 

27 h, w, c = INPUT_SHAPE 

28 (x_train, y_train), (x_test, y_test) = mnist.load_data() 

29 x_train = x_train.reshape(x_train.shape[0], h, w) 

30 x_test = x_test.reshape(x_test.shape[0], h, w) 

31 return list(zip(x_train, y_train)), list(zip(x_test, y_test)) 

32 

33 

34def create_network(): 

35 from keras.models import Sequential 

36 from keras.layers import Dense, Dropout, Flatten 

37 from keras.layers import Conv2D, MaxPooling2D 

38 

39 model = Sequential() 

40 model.add(Conv2D(32, (3, 3), activation='relu', input_shape=INPUT_SHAPE)) 

41 model.add(Conv2D(64, (3, 3), activation='relu')) 

42 model.add(MaxPooling2D(pool_size=(2, 2))) 

43 model.add(Dropout(0.25)) 

44 model.add(Flatten()) 

45 model.add(Dense(128, activation='relu')) 

46 model.add(Dropout(0.5)) 

47 model.add(Dense(NUM_CLASSES, activation='softmax')) 

48 model.compile(loss='categorical_crossentropy', 

49 optimizer='adam', 

50 metrics=['accuracy']) 

51 return KerasNetwork(model, 'cnn_weights.hd5') 

52 

53 

54def train(): 

55 from keras.metrics import categorical_accuracy 

56 

57 print('creating network ...') 

58 network = create_network() 

59 

60 print('loading data...') 

61 train_samples, test_samples = load_samples() 

62 

63 augment = (AugmentImage(0) 

64 .by('identical', 1) 

65 .by('translate', 0.5, [-3, +3], [-3, +3]) 

66 .by('rotate', 0.5, [-5, +5]) 

67 .by('shear', 0.5, [0, 0.2]) 

68 .by('elastic', 0.5, [5, 5], [100, 100], [0, 100])) 

69 transform = (TransformImage(0) 

70 .by('rerange', 0, 255, 0, 1, 'float32')) 

71 build_batch = (BuildBatch(BATCH_SIZE, prefetch=0) 

72 .input(0, 'image', 'float32') 

73 .output(1, 'one_hot', 'uint8', NUM_CLASSES)) 

74 plot = PlotLines((0, 1), layout=(2, 1), every_sec=1) 

75 

76 

77 print('training...', NUM_EPOCHS) 

78 for epoch in range(NUM_EPOCHS): 

79 print('EPOCH:', epoch) 

80 

81 t_loss, t_acc = (train_samples >> PrintProgress(train_samples) >> 

82 Pick(PICK) >> augment >> transform >> Shuffle(100) >> 

83 build_batch >> network.train() >> plot >> Unzip()) 

84 print('train loss : {:.6f}'.format(t_loss >> Mean())) 

85 print('train acc : {:.1f}'.format(100 * (t_acc >> Mean()))) 

86 

87 e_acc = (test_samples >> transform >> build_batch >> 

88 network.evaluate([categorical_accuracy])) 

89 print('test acc : {:.1f}'.format(100 * e_acc)) 

90 

91 network.save_best(e_acc, isloss=False) 

92 

93 

94if __name__ == "__main__": 

95 train()
