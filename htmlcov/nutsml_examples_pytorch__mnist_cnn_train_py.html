28class Model(nn.Module): 

29 """Pytorch model""" 

30 

31 def __init__(self, device='cpu'): 

32 """Construct model on given device, e.g. 'cpu' or 'cuda'""" 

33 super(Model, self).__init__() 

34 

35 self.layers = nn.Sequential( 

36 nn.Conv2d(1, 10, kernel_size=5), 

37 nn.MaxPool2d(2), 

38 nn.ReLU(True), 

39 nn.BatchNorm2d(10), 

40 nn.Conv2d(10, 20, kernel_size=5), 

41 nn.MaxPool2d(2), 

42 nn.ReLU(True), 

43 nn.BatchNorm2d(20), 

44 Reshape(), 

45 nn.Linear(320, 50), 

46 nn.ReLU(True), 

47 nn.Linear(50, 10), 

48 ) 

49 self.to(device) # set device before constructing optimizer 

50 

51 # required properties of a model to be wrapped as PytorchNetwork! 

52 self.device = device # 'cuda', 'cuda:0' or 'gpu' 

53 self.losses = F.cross_entropy # can be list of loss functions 

54 self.optimizer = optim.Adam(self.parameters()) 

55 

56 def forward(self, x): 

57 """Forward pass through network for input x""" 

58 return self.layers(x)

61# definition of nuts 

62BATCHSIZE = 64 

63build_batch = (nm.BuildBatch(BATCHSIZE, verbose=False) 

64 .input(0, 'image', 'float32', True) 

65 .output(1, 'number', 'int64')) 

66build_pred_batch = (nm.BuildBatch(BATCHSIZE, verbose=False) 

67 .input(0, 'image', 'float32', True)) 

68augment = (nm.AugmentImage(0) 

69 .by('identical', 1) 

70 .by('translate', 0.2, [-3, +3], [-3, +3]) 

71 .by('rotate', 0.2, [-30, +30]) 

72 .by('shear', 0.2, [0, 0.2]) 

73 .by('elastic', 0.2, [5, 5], [100, 100], [0, 100]) 

74 ) 

75vec2img = nf.MapCol(0, lambda x: (x.reshape([28, 28]) * 255).astype('uint8'))

78def accuracy(y_true, y_pred): 

79 """Compute accuracy""" 

80 from sklearn.metrics import accuracy_score 

81 y_pred = [yp.argmax() for yp in y_pred] 

82 return 100 * accuracy_score(y_true, y_pred)

85def train(network, x, y, epochs=3): 

86 """Train network for given number of epochs""" 

87 for epoch in range(epochs): 

88 print('epoch', epoch + 1) 

89 losses = (zip(x, y) >> nf.PrintProgress(x) >> vec2img >> 

90 augment >> nf.Shuffle(1000) >> build_batch >> 

91 network.train() >> nf.Collect()) 

92 print('train loss:', np.mean(losses))

95def validate(network, x, y): 

96 """Compute validation/test loss (= mean over batch losses)""" 

97 losses = (zip(x, y) >> nf.PrintProgress(x) >> vec2img >> 

98 build_batch >> network.validate() >> nf.Collect()) 

99 print('val loss:', np.mean(losses))

102def predict(network, x, y): 

103 """Compute network outputs and print accuracy""" 

104 preds = (zip(x, y) >> nf.PrintProgress(x) >> vec2img >> 

105 build_pred_batch >> network.predict() >> nf.Collect()) 

106 acc = accuracy(y, preds) 

107 print('test acc', acc)

109 

110def evaluate(network, x, y): 

111 """Evaluate network performance (here accuracy)""" 

112 metrics = [accuracy] 

113 result = (zip(x, y) >> nf.PrintProgress(x) >> vec2img >> 

114 build_batch >> network.evaluate(metrics)) 

115 return result
